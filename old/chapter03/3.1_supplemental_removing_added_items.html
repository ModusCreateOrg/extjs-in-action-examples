Ext.getCmp('myWin').add({
    title : 'Appended Panel',
    id    : 'addedPanel',
    html  : 'Hello there!'
});

Ext.getCmp('myWin').doLayout();

Ext.getCmp('myWin').insert(1, {
    title : 'Inserted Panel',
    id    : 'insertedPanel',
    html  : 'It is cool here!'
});

Ext.getCmp('myWin').doLayout();

var panel = Ext.getCmp('addedPanel');
Ext.getCmp('myWin').remove(panel);
